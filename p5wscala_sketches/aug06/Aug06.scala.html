33     /** 
34     * Represents an orientation in the world, represented either by rectangular i-j coordinates or by azimuth-elevation coordinates. 
35     * Create Orient objects by calling Orient.fromIJ or Orient.fromAE. 
36     * i ranges from [0, resolution - 1] while j ranges from [0, resolution] (that is, i is exclusive of the resolution endpoint but j is inclusive) 
37     * az ranges from [0, 2*PI*(resolution-1)/resolution], while el ranges from [-PI/2, PI/2] 
38     * azimuth (az) is synonymous with longitiude, while elevation (el) is exactly the same measurement as latitude (using radians) 
39     */ 
40       case class Orient protected[K] (i:Int, j:Int) extends (Int, Int)(i, j) { 
41    
42     //    private def this(aa:Float, ee:Float) = this() 
43     //    val i = _1 
44     //    val j = _2 
45         lazy val i1 = Orient.fromIJ(i+1, j) 
46         lazy val j1 = Orient.fromIJ(i, j+1) 
47         lazy val ij1 = Orient.fromIJ(i+1, j+1) 
48    
49         lazy val az = map(i, 0, resolution, 0, TWO_PI) 
50         lazy val el = map(j, 0, resolution, -PI/2, PI/2) 
51    
52         def +(d:Orient) = Orient.fromIJ(i + d.i, j + d.j) 
53    
54         def offset(azEl:(Float, Float)) = Orient.fromAE(az+azEl._1, el + azEl._2) 
55    
56         def unary_- = Orient.fromIJ(-i, -j) 
57    
58         override def toString = "(i, j = "+i+", "+j+") (az, el = "+az+", "+el+")" 
59       } 
60       object Orient { 
61         private val cache = Array.ofDim[Orient](resolution*(resolution+1)) 
62         (0 to resolution).foreach( j => (0 until resolution).foreach( i => cache(j*resolution+i) = Orient(i, j))) 
63    
64         def fromIJ(ii:Int, jj:Int):Orient = cache( 
65           { val j = jj % (resolution + 1); if(j < 0) j + (resolution + 1) else j } * resolution + 
66           { val i = ii % resolution; if(i < 0) i + resolution else i }) 
67          
68         def fromAE(az:Float, el:Float):Orient = fromIJ(map(az, 0, TWO_PI, 0, resolution).toInt, map(el, -PI/2, PI/2, 0, resolution).toInt) 
69    
70         def all = 
71           (0 to resolution).flatMap( j => 
72             (0 until resolution).map( i => 
73               fromIJ(i, j))) 
74    
75         def rand = fromIJ(random(resolution).toInt, random(resolution+1).toInt) 
76       } 
77     }

80   /** Grid holds the actual underlying arrays representing the planet's heights and colors. The underlying array is set up 
81   * in the same way that Processing's pixel array is. You may imagine the planet as being a 2D image of width = resolution and 
82   * height = resolution + 1, where the i and j values in an Orient represent the x and y coordinates of the image. Except that 
83   * Grid holds both height and color values for each pixel. 
84               i (az) 
85         +------------------+   ^ 
86         |                  |   | 
87         |                  |   |height 
88         |                  |   | 
89     j   |                  |   |   = 
90         |                  |   | 
91    (ele)|                  |   |resolution+1 
92         |                  |   | 
93         |                  |   | 
94         |                  |   | 
95         +------------------+   v 
96    
97         |------------------| 
98          width = resolution 
99    
100  */ 
101    private object Grid { 
102      private val grid = Array.ofDim[Float](resolution*(resolution+1)) 
103      private val colors = Array.ofDim[Int](resolution*(resolution+1)) 
104   
105      def reset() { 
106        (0 until grid.length).foreach(i => {grid(i) = 0f; colors(i) = 0; minCache = 0;}) 
107      } 
108   
109      def update(o: Orient, x:Float) { grid.update(index(o), x); updateCache(x); } 
110      def update(o: Orient, c:Int) { colors.update(index(o), c) } 
111   
112      def heightAt(o: Orient) = grid(index(o)) 
113      def colorAt(o: Orient) = colors(index(o)) 
114   
115      /** 
116      *  Calculate the magnitude of the gradient at the given orient. This is calculated by calculating the partial 
117      *  derivative with respect to the i and j directions (averaging the derivatives of both plus and minus in each direction) 
118      *  and then using the pythagorean theorem with the two average-partials. 
119      */ 
120      def slopeAt(o:Orient) = { 
121        val myHeight = heightAt(o) 
122        def slope(di:Int, dj:Int) = heightAt(o + (di, dj)) - myHeight // /dist(0, 0, di, dj) -- commented out because it's always 1 
123        def avgSlope(di:Int, dj:Int) = (slope(di, dj) + slope(-di, -dj)) / 2 
124        val (partialI, partialJ) = (avgSlope(1, 0), avgSlope(0, 1)) 
125        dist(0, 0, partialI, partialJ) 
126      } 
127   
128      private var minCache = Float.MaxValue 
129      private var maxCache = 0f 
130      private def updateCache(newVal:Float) { 
131  //      mhCache = grid.reduceLeft(_ min _) //todo: put back in! 
132        minCache = minCache min newVal 
133        maxCache = maxCache max newVal 
134      } 
135   
136      def minHeight = minCache 
137      def maxHeight = maxCache 
138    }

150    * The colorizerDefinedAt method should return a boolean indicating whether the colorizer will make any noticable changes

163   
164      def colorizerDefinedAt(h: Orient):Boolean

223    * A Feature changes the landscape in some way; they are mountains, hills, craters, and also include features generated 
224    * through noise. 
225    * The apply method should denote how much height the Feature has at the given Orient. 
226    * The isDefinedAt method should denote whether the feature exists or makes any noticable difference at the given Orient.

228    sealed private abstract class Feature extends PartialFunction[Orient, Float] { 
229      def addGrid() { updateGrid(_ + _) } 
230      def overwriteGrid() { updateGrid((a, b) => b) } 
231      def updateGrid(combiner: (Float, Float) => Float) {//combiner specifies "how" to add this feature to the Grid 
232        Orient.all.filter(isDefinedAt _).foreach(orient => Grid(orient) = combiner(Grid.heightAt(orient), apply(orient))) 
233      } 
234   
235      def +(f:Feature) = new FeatureCombiner(this, f, _ + _) 
236    }

238      def apply(v1: Orient) = func(f1(v1), f2(v1)) 
239      def isDefinedAt(x: Orient) = f1.isDefinedAt(x) && f2.isDefinedAt(x) 
240    }

280    /** 
281    * A CenteredFeature is a feature that has some sort of focal point; e.g. the tip of a mountain, the bottom of a crater. 
282    * As of now, CenteredFeature also has a <radius> field to denote maximum outward extent from the <center> of the feature; 
283    * in this way, the feature is only defined within a circle of center <center> and radius <radius>. The <radius> is defined 
284    * in AE coordinates (so as to keep resolution-independence). 
285    */ 
286    private abstract class CenteredFeature(center:Orient, radius:Float) extends Feature { 
287   
288      /** 
289      * Return the height of the CenteredFeature at the given polar coordinate away from the center. "rad" is a normalized 
290      * metric in the range [0, 1] that maps to [0, radius] away from "center" 
291      */ 
292      def apply(rad:Float, ang:Float):Float 
293   
294      /** 
295      * Returns the *real* offset in AE coordinates from the center to the given Orient. This method 
296      * returns small negative numbers to denote that it's easier to go "backwards". Originally there was 
297      * a.-(b), where a and b were orients, but since orients are always positive, it'd say things like 
298      * (1, 1) - (5, 5) = (96, 97), which is technically true but which gives you junk values when used for 
299      * distance. 
300      * This method returns (change in Azimuth, change in Elevation). 
301      */ 
302      def getOffset(x: Orient): (Float, Float) = { 
303        import Methods.distance 
304        val dAz = distance(center.az, x.az, TWO_PI); 
305        val dEl = distance(center.el + PI / 2, x.el + PI / 2, PI + .001f);//add an epsilon value to the wrap to make +PI inclusive 
306        (dAz, dEl) 
307      } 
308   
309      /** 
310      * Return the height of the CenteredFeature at the given Orient. This method should just convert orients to polar 
311      * coordinates centered around "center". 
312      */ 
313      final def apply(x: Orient) = { 
314        val offset = getOffset(x) 
315        val angle = atan2(offset._2, offset._1) //atan2(y, x) 
316        val rad = map(dist(0, 0, offset._2, offset._1), 0, radius, 0, 1) 
317        apply(rad, angle) 
318      } 
319   
320      def isDefinedAt(x: Orient) = { 
321        val offset = getOffset(x) 
322        dist(0, 0, offset._1, offset._2) < radius 
323      } 
324    }

330    private object PlateauFeature { 
331      private var offsetStart = 0f 
332   
333      def findGoodOffset(range:Float = TWO_PI, epsilon:Float = 1e-5f, start:Float = offsetStart) = { 
334        val sampleInterval = .001f; //maybe calculate this with respect to epsilon 
335        var offset = start; 
336        while(abs(noise(offset)-noise(offset+range)) > epsilon) offset += sampleInterval 
337        offset 
338      } 
339   
340      //Returns a Float => Float that takes an angle as a parameter and returns the outer radius of the plateau at that angle. 
341      //The returned radius should be in the range [0, 1] 
342      def nextFunction() = { 
343        val offset = findGoodOffset() 
344        offsetStart = offset //this ensures that no two plateaus will be the same. 
345        (angle:Float) => noise(angle + offset) 
346      } 
347   
348    }

351      val func = PlateauFeature.nextFunction() 
352   
353      def apply(rad: Float, ang: Float) = if(func(ang) > rad) height else 0 
354    }

367      /** 
368      * range is a mapping between the grid's height and its color. calling range.get(seaLevel) gives you the color of 
369      * land of height seaLevel (aka depth zero). range.get(0) gives you the surface color for land at a height of 0 (aka depth 
370      * = seaLevel) 
371      */ 
372      val map = { 
373        val grad = new ColorGradient()

375        * Add a hex color to the given depth. depthNorm ranges from [0, 1] and is a normalized metric for the depth of the water.

378        * addColorAt(1, "02184F") means that at a depth of seaLevel, the color will be 02184F (dark blue) 
379        */ 
380        def addColorAt(depthNorm:Float, hex:String) = grad.addColorAt(depthNorm*seaLevel, TColor.newHex(hex).setAlpha(1)) 
381        addColorAt(1, "FAF7A5") 
382        addColorAt(.95f, "3242D1") 
383        addColorAt(.8f, "222E9C") 
384        addColorAt(.5f, "101B82") 
385        addColorAt(0, "02184F") 
386   
387        grad.calcGradient() 
388      }
