52     def phasor(amp: Float, theta: Float) = Vec2.fromPolar(amp, theta);

54     /* 
55       a) make phasors only call eval on some already parsed Expr; we will need to have three Expr vars for amp, freq, and phase. 
56         i) think about threading issues. 
57       b) In CP5, reparse the strings and update the Expr vars every loop. Use CP5 as a mutex to remove threading issues 
58      */ 
59     //amplitude, frequency, phase; y(t) = Ae^(i*2*pi*f*t + phase) 
60     def phasors(time:Float = time):Seq[Vec2] = CP5.synchronized{ //Since each PApplet lives on its own thread we're not actually sure when a draw will occur. 
61       (ampExpr, freqExpr, phaseExpr) match { 
62         case (Right(amp), Right(freq), Right(phase)) => (1 to CP5.N).map(n => { 
63           varMap("n") = n; 
64           phasor(amp.eval, 2*PI*freq.eval*time + phase.eval) 
65         }) 
66         case _ => List() 
67       } 
68     } 
69    
70     def totalPhasor(time:Float = time) = (phasors(time) foldLeft Vec2())(_ + _)

181    private object CosGraph extends MyPApplet with Grapher { 
182   
183      override def draw() { 
184        import Argand.cam 
185        background(bgColor); 
186   
187        /* 
188         * Here's how things will connect: 
189         * a) CosGraph's drawing methods interpret all numbers as screen coordinates. 
190         * b) To draw something as if it were living in model space, we provide a method to convert from 
191         *    model coordinates to screen coordinates - this is the toScreen method. 
192         * c) For all drawing invocations, we use toScreen(model) as necessary 
193         * 
194         * The model space has an 
195         * 
196         * x-range defined as [cam.modelX(0), cam.modelX(width)]    (scale = cam.getScale) 
197         * y-range defined as [-scaleRange, scaleRange]             (scale = height / (2*scaleRange)) 
198         */ 
199   
200        def toScreen(model:(Float, Float)) = Vec2(map(model._1, cam.modelX(0), cam.modelX(width), 0, width), 
201                                                  map(model._2, -scaleRange, scaleRange, 0, height)); 
202   
203   
204        //draw coordinate lines 
205        strokeWeight(1); 
206        stroke(0); 
207        line(toScreen(0, -scaleRange), toScreen(0, scaleRange)) //draw a vertical line down the middle of the screen 
208        line(0, height/2, width, height/2) //draw a horizontal line through the middle of the screen 
209   
210        //draw the big current line 
211        strokeWeight(6); 
212        stroke(cosColor); 
213        line(toScreen(0, 0), toScreen(totalPhasor().x, 0f)); 
214   
215        //draw the graph. We want height/2 to call totalPhasor(time) exactly. We want the graph to extend up and down by scaleRange. 
216        //We should iterate through each y coordinate, transform it to the corresponding time 
217        //(y = 0 -> t = time - tRange, y = height/2 -> t = time, y = height -> t = time + tRange). 
218        stroke(darken(cosColor)); noFill(); 
219        strokeWeight(1); 
220        beginShape(); 
221        for(y <- 0 until height) { 
222          val timeOffset = (y - height / 2) * scaleRange / height; 
223          val curTime = time + timeOffset; 
224          val x = totalPhasor(curTime).x 
225          vertex(toScreen(x, 0).x, y); //we only need the X coordinate in the model space 
226        } 
227        endShape(); 
228      } 
229    }

231    private object SinGraph extends MyPApplet with Grapher { 
232   
233      override def draw() { 
234        import Argand.cam 
235        background(bgColor); 
236   
237        /* 
238         * Here's how things will connect: 
239         * a) SinGraph's drawing methods interpret all numbers as screen coordinates. 
240         * b) To draw something as if it were living in model space, we provide a method to convert from 
241         *    model coordinates to screen coordinates - this is the toScreen method. 
242         * c) For all drawing invocations, we use toScreen(model) as necessary 
243         * 
244         * The model space has an 
245         * 
246         * x-range defined as [-scaleRange, scaleRange]             (scale = width / (2 * scaleRange)) 
247         * y-range defined as [cam.modelY(0), cam.modelY(height)]   (scale = cam.getScale) 
248         */ 
249   
250        def toScreen(model:(Float, Float)) = Vec2(map(model._1, -scaleRange, scaleRange, 0, width), 
251                                                  map(model._2, cam.modelY(0), cam.modelY(height), 0, height)); 
252   
253   
254  //draw coordinate lines 
255        strokeWeight(1); 
256        stroke(0); 
257        //draw a vertical line down the center of the screen 
258        line(width/2, 0, width/2, height) 
259        //draw a horizontal line through model-y zero 
260        line(toScreen(-scaleRange, 0), toScreen(scaleRange, 0)) 
261   
262        //draw the big current line 
263        strokeWeight(6) 
264        stroke(sinColor) 
265        line(toScreen(0, 0), toScreen(0f, -totalPhasor().y)) //negate 
266   
267        //draw the graph. Go from x [0, width], transforming it to the corresponding time. 
268        stroke(darken(sinColor)); noFill(); 
269        strokeWeight(1); 
270        beginShape(); 
271        for(x <- 0 until width) { 
272          val timeOffset = (x - width/2) * (scaleRange / width); 
273          val curTime = time + timeOffset; 
274          val y = totalPhasor(curTime).y; 
275          vertex(x, toScreen(0, -y).y); //we only need the Y coordinate in the model space; //negate 
276        } 
277        endShape(); 
278      } 
279    }

281    object Argand extends MyPApplet { 
282   
283      import PConstants._; 
284   
285      lazy val cam = new zhang.Camera(this); 
286   
287      override def setup() { 
288        size(250, 250) 
289        cam.setCenter(0, 0); 
290        cam.setViewportWidth(4); 
291        smooth(); 
292        noLoop(); 
293        Thread.currentThread().setName("Argand Thread") 
294      } 
295   
296      override def draw() { 
297        background(bgColor); 
298        zhang.Methods.strokeWeightScreen(this, 1, cam); 
299        stroke(0); 
300        noFill(); 
301        ellipse(0, 0, 2, 2); //unit circle of diameter 2 = radius 1 
302        P5Util.drawGraphAxes(this, cam); 
303   
304        zhang.Methods.strokeWeightScreen(this, 6f, cam); 
305        val tp = totalPhasor() 
306        stroke(cosColor); 
307        line(0, 0, tp.x, 0) 
308        stroke(sinColor); 
309        line(0, 0, 0, -tp.y) //negate 
310   
311        zhang.Methods.strokeWeightModel(this, 2f, cam); 
312        stroke(vecColor); 
313        (phasors() foldLeft Vec2()) { 
314          (accum: Vec2, enum: Vec2) => { 
315            def drawPhasor(start: Vec2, p: Vec2) { 
316              def line(a:Vec2, b:Vec2) { Argand.line(a.x, -a.y, b.x, -b.y); } //negate 
317              line(start, p); 
318              val offset = p - start; 
319              line(p, p - offset.rotate(PI / 8).scale(.2f)) 
320              line(p, p - offset.rotate(-PI / 8).scale(.2f)) 
321            } 
322            drawPhasor(accum, accum + enum); 
323            accum + enum; 
324          } 
325        } 
326      } 
327   
328      override def mouseDragged() { 
329        cam.translate(pmouseX - mouseX, pmouseY - mouseY) 
330      }
